public class JuegoFacade {
    Tablero tablero;
    Pieza pieza;

    BienvenidaJuego bienvenidaJuego;

    public JuegoFacade(){
        tablero = new Tablero();
        pieza = new Pieza();
        bienvenidaJuego = new BienvenidaJuego();

    }

    public void elegirPos(int x, int y, char figura){
        if(figura == 'a'){
            for (int i = x; i<9; i++){
